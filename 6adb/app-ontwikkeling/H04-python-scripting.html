1. Python basics recap

Voordat we aan data-analyse beginnen, herhalen we de essentie van Python:

# Lijsten en dictionary's
producten = ["Laptop", "Muis", "Monitor"]
prijzen   = {"Laptop": 999, "Muis": 25, "Monitor": 349}

# List comprehension (compacte for-loop)
dure = [p for p in producten if prijzen[p] > 100]
print(dure)  # ['Laptop', 'Monitor']

# Functies
def bereken_btw(prijs, tarief=0.21):
    return round(prijs * (1 + tarief), 2)

print(bereken_btw(999))       # 1208.79
print(bereken_btw(25, 0.06))  # 26.5

# Lambda (anonieme functie)
sorteer_prijs = lambda item: item[1]
gesorteerd = sorted(prijzen.items(), key=sorteer_prijs)
print(gesorteerd)  # [('Muis', 25), ('Monitor', 349), ('Laptop', 999)]

2. Bestanden lezen en schrijven

Python kan eenvoudig bestanden lezen en schrijven via de ingebouwde open() functie.

# Tekstbestand schrijven
with open("namen.txt", "w", encoding="utf-8") as bestand:
    bestand.write("Alice\n")
    bestand.write("Bob\n")
    bestand.write("Charlie\n")

# Tekstbestand lezen (alle regels)
with open("namen.txt", "r", encoding="utf-8") as bestand:
    regels = bestand.readlines()
    for regel in regels:
        print(regel.strip())

# CSV-bestand schrijven
import csv

data = [
    ["naam",   "prijs", "voorraad"],
    ["Laptop",   999,      10],
    ["Muis",      25,      50],
    ["Monitor",  349,      15]
]

with open("producten.csv", "w", newline="", encoding="utf-8") as f:
    schrijver = csv.writer(f)
    schrijver.writerows(data)
Tip — with-statement

Het with-statement zorgt er automatisch voor dat het bestand gesloten wordt, ook als er een fout optreedt. Gebruik altijd with open(...) in plaats van open() zonder with.

3. pandas

pandas is de standaardbibliotheek voor data-analyse in Python. Het werkt met DataFrames: tabellen met rijen en kolommen, vergelijkbaar met een Excel-spreadsheet.

# Installeren (eenmalig)
# pip install pandas

import pandas as pd

# DataFrame aanmaken vanuit dictionary
df = pd.DataFrame({
    "naam":     ["Alice", "Bob", "Charlie", "Dana"],
    "leeftijd": [22, 35, 28, 42],
    "afdeling": ["IT", "HR", "IT", "Verkoop"],
    "salaris":  [3200, 2800, 3500, 3100]
})

3.1 CSV inlezen en verkennen

# CSV inlezen
df = pd.read_csv("verkoop.csv", encoding="utf-8")

# Eerste 5 rijen bekijken
print(df.head())

# Basisstatistieken
print(df.describe())

# Kolomnamen
print(df.columns.tolist())

# Aantal rijen en kolommen
print(df.shape)      # (100, 5) = 100 rijen, 5 kolommen

# Ontbrekende waarden opsporen
print(df.isnull().sum())

3.2 Filteren, selecteren en groeperen

# Kolom selecteren
namen = df["naam"]

# Rijen filteren
it_medewerkers = df[df["afdeling"] == "IT"]
hogeSalarissen = df[df["salaris"] > 3000]

# Meerdere condities
it_en_hoog = df[(df["afdeling"] == "IT") & (df["salaris"] > 3000)]

# Groeperen en aggregeren
per_afdeling = df.groupby("afdeling")["salaris"].agg(["mean", "min", "max"])
print(per_afdeling)
#            mean   min   max
# afdeling
# HR         2800  2800  2800
# IT         3350  3200  3500
# Verkoop    3100  3100  3100

# Nieuwe kolom berekenen
df["btw"] = df["prijs"] * 0.21
df["totaal"] = df["prijs"] + df["btw"]

4. matplotlib: grafieken

matplotlib laat toe grafieken te maken vanuit Python-data.

# Installeren: pip install matplotlib
import matplotlib.pyplot as plt
import pandas as pd

# Data
df = pd.read_csv("verkoop.csv")
per_maand = df.groupby("maand")["omzet"].sum()

# Lijndiagram
plt.figure(figsize=(10, 5))
plt.plot(per_maand.index, per_maand.values, marker="o", color="#6C63FF")
plt.title("Maandelijkse omzet 2025")
plt.xlabel("Maand")
plt.ylabel("Omzet (EUR)")
plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.savefig("omzet_per_maand.png", dpi=150)
plt.show()

# Staafdiagram
per_categorie = df.groupby("categorie")["aantal"].sum()

fig, assen = plt.subplots(1, 2, figsize=(12, 5))

# Links: staafdiagram
assen[0].bar(per_categorie.index, per_categorie.values, color=["#6C63FF","#FF6584","#43E97B"])
assen[0].set_title("Verkoop per categorie")
assen[0].set_xlabel("Categorie")
assen[0].set_ylabel("Aantal")

# Rechts: taartdiagram
assen[1].pie(per_categorie.values, labels=per_categorie.index, autopct="%1.1f%%")
assen[1].set_title("Aandeel per categorie")

plt.tight_layout()
plt.savefig("categorieanalyse.png", dpi=150)
plt.show()
