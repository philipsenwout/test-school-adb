🎯 Leerdoelen
  • De leerling kan pd.read_sql() gebruiken om queryresultaten direct in een DataFrame te laden.
  • De leerling kan groupby().agg() toepassen voor meerdere aggregaties tegelijk.
  • De leerling kan plt.subplots() gebruiken om een grid van grafieken op te bouwen.
  • De leerling kan lijn-, staaf-, horizontale staaf- en taartgrafieken maken met matplotlib.
  • De leerling kan een volledig dashboard exporteren als PNG-bestand.

1. pandas DataFrame van een query-resultaat

pandas heeft een ingebouwde functie om direct een DataFrame te maken vanuit een databasequery via pd.read_sql():

import pandas as pd
import mysql.connector
import matplotlib.pyplot as plt

# Verbinding maken
verbinding = mysql.connector.connect(
    host="localhost", user="root",
    password="geheim", database="winkel_db"
)

# Direct een DataFrame laden vanuit een query
df_bestellingen = pd.read_sql("""
    SELECT
        b.bestelling_id,
        k.naam AS klant,
        k.stad,
        b.besteldatum,
        SUM(br.aantal * p.prijs) AS totaal
    FROM bestellingen b
    JOIN klanten k             ON b.klant_id = k.klant_id
    JOIN bestelling_regels br  ON b.bestelling_id = br.bestelling_id
    JOIN producten p            ON br.product_id = p.product_id
    GROUP BY b.bestelling_id, k.naam, k.stad, b.besteldatum
""", verbinding)

verbinding.close()

print(df_bestellingen.head())
print(df_bestellingen.dtypes)  # datatypes controleren
Tip — pd.read_sql vs. cursor.fetchall()

pd.read_sql() is de eenvoudigste manier: het voert de query uit en laadt het resultaat direct in een DataFrame. Alternatieven: cursor.fetchall() gevolgd door pd.DataFrame(rijen). Beide werken, maar pd.read_sql() is compacter.

2. groupby en aggregatie in pandas

# Datumkolom omzetten naar datetime
df_bestellingen["besteldatum"] = pd.to_datetime(df_bestellingen["besteldatum"])

# Maand extracteren
df_bestellingen["maand"] = df_bestellingen["besteldatum"].dt.to_period("M")

# Omzet per maand
omzet_per_maand = df_bestellingen.groupby("maand")["totaal"].sum().reset_index()
print(omzet_per_maand)

# Meerdere aggregaties tegelijk
analyse = df_bestellingen.groupby("stad").agg(
    bestellingen = ("bestelling_id", "count"),
    totaal_omzet = ("totaal",        "sum"),
    gem_bedrag   = ("totaal",        "mean"),
    max_bedrag   = ("totaal",        "max")
).reset_index()

analyse["gem_bedrag"] = analyse["gem_bedrag"].round(2)
print(analyse.sort_values("totaal_omzet", ascending=False))

5. Volledig dashboard script

import pandas as pd
import mysql.connector
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches

def laad_data():
    """Verbinding maken en data ophalen als DataFrames."""
    conn = mysql.connector.connect(
        host="localhost", user="root",
        password="geheim", database="winkel_db"
    )
    df_orders = pd.read_sql("""
        SELECT b.bestelling_id,
               k.naam AS klant, k.stad,
               b.besteldatum,
               p.categorie,
               SUM(br.aantal * p.prijs) AS totaal
        FROM bestellingen b
        JOIN klanten k ON b.klant_id = k.klant_id
        JOIN bestelling_regels br ON b.bestelling_id = br.bestelling_id
        JOIN producten p ON br.product_id = p.product_id
        GROUP BY b.bestelling_id, k.naam, k.stad, b.besteldatum, p.categorie
    """, conn)
    conn.close()
    df_orders["besteldatum"] = pd.to_datetime(df_orders["besteldatum"])
    df_orders["maand"] = df_orders["besteldatum"].dt.month
    return df_orders

def maak_dashboard(df):
    """Dashboard met 4 grafieken opbouwen."""
    fig, assen = plt.subplots(2, 2, figsize=(14, 10))
    fig.suptitle("Verkoopdashboard", fontsize=18, fontweight="bold", y=1.02)

    # 1. Omzet per maand
    per_maand = df.groupby("maand")["totaal"].sum()
    assen[0,0].plot(per_maand.index, per_maand.values,
                    marker="o", color="#6C63FF")
    assen[0,0].set_title("Maandelijkse omzet")

    # 2. Omzet per categorie
    per_cat = df.groupby("categorie")["totaal"].sum().sort_values(ascending=False)
    assen[0,1].bar(per_cat.index, per_cat.values, color="#FF6584")
    assen[0,1].set_title("Omzet per categorie")
    assen[0,1].tick_params(axis="x", rotation=30)

    # 3. Taartdiagram steden
    per_stad = df.groupby("stad")["totaal"].sum()
    assen[1,0].pie(per_stad, labels=per_stad.index, autopct="%1.1f%%")
    assen[1,0].set_title("Omzet per stad")

    # 4. Top-10 klanten
    top_klanten = df.groupby("klant")["totaal"].sum().nlargest(10)
    assen[1,1].barh(top_klanten.index, top_klanten.values, color="#43E97B")
    assen[1,1].set_title("Top 10 klanten")

    plt.tight_layout()
    plt.savefig("dashboard.png", dpi=150, bbox_inches="tight")
    print("Dashboard opgeslagen als dashboard.png")

# Uitvoeren
df = laad_data()
maak_dashboard(df)

Samenvatting

Samenvatting
  • pd.read_sql(query, verbinding) laadt queryresultaten direct in een DataFrame.
  • groupby("kolom").agg({...}) groepeert en aggregeert meerdere kolommen tegelijk.
  • plt.subplots(r, k) maakt een r×k grid van grafieken. Toegang via assen[rij, kolom].
  • Grafiektypes: plot() (lijn), bar() (staaf), barh() (horizontaal), pie() (taart).
  • plt.savefig("bestand.png", dpi=150) exporteert de grafiek.
