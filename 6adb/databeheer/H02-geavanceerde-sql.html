1. Testschema

We werken in dit hoofdstuk met een e-commerce schema:

CREATE TABLE klanten (
  klant_id INT AUTO_INCREMENT PRIMARY KEY,
  naam     VARCHAR(100),
  stad     VARCHAR(50)
);
CREATE TABLE producten (
  product_id INT AUTO_INCREMENT PRIMARY KEY,
  naam       VARCHAR(100),
  prijs      DECIMAL(10,2),
  categorie  VARCHAR(50)
);
CREATE TABLE bestellingen (
  bestelling_id INT AUTO_INCREMENT PRIMARY KEY,
  klant_id      INT,
  besteldatum   DATE,
  FOREIGN KEY (klant_id) REFERENCES klanten(klant_id)
);
CREATE TABLE bestelling_regels (
  regel_id      INT AUTO_INCREMENT PRIMARY KEY,
  bestelling_id INT,
  product_id    INT,
  aantal        INT,
  FOREIGN KEY (bestelling_id) REFERENCES bestellingen(bestelling_id),
  FOREIGN KEY (product_id)    REFERENCES producten(product_id)
);

2. GROUP BY en HAVING

GROUP BY groepeert rijen met dezelfde waarde in een kolom. HAVING filtert na groepering (zoals WHERE maar dan op geaggregeerde waarden).

-- Aantal bestellingen per klant
SELECT k.naam, COUNT(b.bestelling_id) AS aantal_bestellingen
FROM klanten k
JOIN bestellingen b ON k.klant_id = b.klant_id
GROUP BY k.klant_id, k.naam
ORDER BY aantal_bestellingen DESC;

-- Enkel klanten met meer dan 2 bestellingen (HAVING)
SELECT k.naam, COUNT(b.bestelling_id) AS aantal
FROM klanten k
JOIN bestellingen b ON k.klant_id = b.klant_id
GROUP BY k.klant_id, k.naam
HAVING COUNT(b.bestelling_id) > 2;

-- Totale omzet per categorie
SELECT p.categorie,
       SUM(br.aantal * p.prijs) AS omzet,
       AVG(p.prijs)             AS gemiddelde_prijs,
       COUNT(DISTINCT p.product_id) AS aantal_producten
FROM producten p
JOIN bestelling_regels br ON p.product_id = br.product_id
GROUP BY p.categorie
HAVING omzet > 1000
ORDER BY omzet DESC;
WHERE vs. HAVING

WHERE filtert vóór groepering (op individuele rijen). HAVING filtert na groepering (op gegroepeerde resultaten). Je kan beide combineren in dezelfde query.

3. Subqueries

Een subquery is een SELECT-statement binnenin een ander statement. Ze kan voorkomen in de WHERE, FROM of SELECT clausule.

-- Producten duurder dan het gemiddelde (scalar subquery)
SELECT naam, prijs
FROM producten
WHERE prijs > (SELECT AVG(prijs) FROM producten);

-- Klanten die iets besteld hebben (IN met subquery)
SELECT naam
FROM klanten
WHERE klant_id IN (
  SELECT DISTINCT klant_id FROM bestellingen
);

-- Klanten die NIETS besteld hebben (NOT IN)
SELECT naam
FROM klanten
WHERE klant_id NOT IN (
  SELECT DISTINCT klant_id FROM bestellingen
);

-- EXISTS: efficiënter dan IN voor grote datasets
SELECT k.naam
FROM klanten k
WHERE EXISTS (
  SELECT 1 FROM bestellingen b
  WHERE b.klant_id = k.klant_id
    AND b.besteldatum > '2025-01-01'
);

4. JOINs verdieping

JOIN typeGeeft terug
INNER JOINEnkel rijen die in béide tabellen overeenkomen
LEFT JOINAlle rijen uit linker tabel + overeenkomsten rechts (NULL als geen match)
RIGHT JOINAlle rijen uit rechter tabel + overeenkomsten links
 
-- INNER JOIN: klanten mét bestellingen
SELECT k.naam, b.besteldatum
FROM klanten k
INNER JOIN bestellingen b ON k.klant_id = b.klant_id;

-- LEFT JOIN: ALLE klanten, ook zonder bestelling
SELECT k.naam, COUNT(b.bestelling_id) AS bestellingen
FROM klanten k
LEFT JOIN bestellingen b ON k.klant_id = b.klant_id
GROUP BY k.klant_id, k.naam;

-- Meervoudige JOIN (3 tabellen)
SELECT k.naam, p.naam AS product, br.aantal
FROM klanten k
JOIN bestellingen b    ON k.klant_id  = b.klant_id
JOIN bestelling_regels br ON b.bestelling_id = br.bestelling_id
JOIN producten p       ON br.product_id = p.product_id
ORDER BY k.naam, p.naam;

5. Aliassen

Aliassen maken lange kolomnamen korter en maken berekende kolommen leesbaar:

-- Kolomalias
SELECT naam AS productnaam,
       prijs AS verkoopprijs,
       prijs * 1.21 AS prijs_incl_btw
FROM producten;

-- Tabel-alias (verplicht bij self-join)
SELECT w1.naam AS werknemer, w2.naam AS manager
FROM werknemers w1
JOIN werknemers w2 ON w1.manager_id = w2.werknemer_id;

6. Views

Een view is een opgeslagen query die je kan aanspreken als een gewone tabel. Handig voor complexe queries die je vaak hergebruikt.

-- View aanmaken
CREATE VIEW bestellingen_overzicht AS
SELECT
  b.bestelling_id,
  k.naam AS klant_naam,
  b.besteldatum,
  SUM(br.aantal * p.prijs) AS totaal_bedrag
FROM bestellingen b
JOIN klanten k             ON b.klant_id = k.klant_id
JOIN bestelling_regels br  ON b.bestelling_id = br.bestelling_id
JOIN producten p            ON br.product_id = p.product_id
GROUP BY b.bestelling_id, k.naam, b.besteldatum;

-- View gebruiken alsof het een tabel is
SELECT * FROM bestellingen_overzicht
WHERE totaal_bedrag > 100
ORDER BY besteldatum DESC;

-- View verwijderen
DROP VIEW IF EXISTS bestellingen_overzicht;

Samenvatting

Samenvatting
  • GROUP BY groepeert rijen. Gebruik aggregatiefuncties: COUNT, SUM, AVG, MIN, MAX.
  • HAVING filtert na groepering; WHERE filtert vóór groepering.
  • Subqueries: scalar (geeft één waarde), IN, EXISTS.
  • INNER JOIN = intersectie; LEFT JOIN = alle rijen links + NULL rechts.
  • Views slaan queries op voor hergebruik: CREATE VIEW naam AS SELECT ....
